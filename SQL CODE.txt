SELECT COUNT(*) AS TOTAL
FROM STUDENTS
INNER JOIN STATES ON STUDENTS.STATEID=STATES.StateID
WHERE STATES.STATEID=6

2.
SELECT  COUNT(SA.stickyID) AS ASSESSMENTS
FROM Stripe_Transactions ST
INNER JOIN STICKY_AUTHOR SA ON ST.UserID = SA.UserID

3.
SELECT count(*) 
FROM Stripe_Transactions ST 
INNER JOIN STICKY_AUTHOR SA ON ST.USERID = SA.UserID WHERE ST.Usertype = 'parent' 
AND ST.ValidityBegin >= DATE_FORMAT(CURRENT_DATE() - INTERVAL 1 MONTH,'%Y-%m-01') 
AND ST.ValidityBegin < DATE_FORMAT(CURRENT_DATE(), '%Y-%m-01')
